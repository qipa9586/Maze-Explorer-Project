## 核心数据结构

### maze.h — 全部声明

```c
#ifndef MAZE_H
#define MAZE_H
#include <stdbool.h>
#include <raylib.h>

// 窗口和迷宫尺寸
#define LENGTH 1600
#define WIDTH 1200
#define CELL_SIZE 30
#define ROWS 40
#define COLS 52

// 居中偏移（自动计算，不改数字就能自适应）
#define OFFSET_X ((LENGTH - COLS * CELL_SIZE) / 2)
#define OFFSET_Y ((WIDTH - ROWS * CELL_SIZE) / 2)
#define WALL_COLOR (Color) {200, 200, 200, 255}
#define BG_COLOR (Color) {30, 30, 30, 255}
#define FONT_PATH "C:/Windows/Fonts/FiraCode-Bold.ttf"

// 链队节点：BFS 用
typedef struct QUEUE {
    int row, col;
    int prevRow, prevCol;   // 从哪个格子来的（队内用）
    struct QUEUE *next;
} QueueNode;

// 链队：头出尾进
typedef struct LinkedQueue {
    QueueNode *front;
    QueueNode *rear;
} Queue;

// 链栈节点：DFS 生成用
typedef struct POSITION {
    int col, row;
} Position;

typedef struct STACK {
    Position pos;
    struct STACK *next;
} StackNode;

// 单个格子：四面墙 + 两个标记
typedef struct CELL {
    bool top, right, left, bottom;   // true = 有墙挡着
    bool visited;                    // DFS 生成和 BFS 寻路共用
    bool isPath;                     // true = 在最短路径上
} Cell;

// 迷宫整体
typedef struct MAZE {
    int rows, cols;
    Cell **grid;           // 动态二维格子数组
    Position *path;        // BFS 回溯出的路径坐标
    int pathLen;           // 路径长度
} Maze;

// 函数声明
void generateRandomizedMaze(Maze *maze);
void mazeSolver(Maze *maze);
Maze *createMaze(int rows, int cols);
void destroyMaze(Maze *maze);
void drawMaze(Maze *maze, Font font);
#endif
```

**设计要点**：

- `OFFSET_X`/`OFFSET_Y` 用了宏表达式而非硬编码数字，换格子尺寸或窗口大小时自动重新计算
- `Position *path` 按最大可能长度 `rows × cols` 分配，保证不越界
- `isPath` 字段放在 Cell 里，GUI 画的时候直接判断，不需要额外的数据结构传递
- 链队（Queue）用 `front`+`rear` 双指针，入队 O(1) 尾插，出队 O(1) 头删

## 一、迷宫创建与销毁 — maze.c

```c
#include <stdlib.h>
#include "maze.h"

Maze *createMaze(int rows, int cols) {
    // 1. 分配迷宫结构体
    Maze *maze = (Maze *)malloc(sizeof(Maze));
    maze->rows = rows;
    maze->cols = cols;

    // 2. 分配二维格子数组（先分行指针，再每行分配 cols 个 Cell）
    maze->grid = (Cell **)malloc(sizeof(Cell *) * maze->rows);
    for (int i = 0; i < maze->rows; i++) {
        maze->grid[i] = (Cell *)malloc(sizeof(Cell) * maze->cols);
    }

    // 3. 分配路径数组（最大可能长度 = 走遍所有格子）
    maze->path = (Position *)malloc(sizeof(Position) * maze->rows * maze->cols);
    maze->pathLen = 0;

    // 4. 初始化每个格子：四面有墙、未访问、不在路径上
    for (int i = 0; i < maze->rows; i++) {
        for (int j = 0; j < maze->cols; j++) {
            maze->grid[i][j].top = true;
            maze->grid[i][j].left = true;
            maze->grid[i][j].right = true;
            maze->grid[i][j].bottom = true;
            maze->grid[i][j].visited = false;
            maze->grid[i][j].isPath = false;
        }
    }
    return maze;
}

void destroyMaze(Maze *maze) {
    // 释放顺序：先每行 → 行指针数组 → path 数组 → 迷宫结构体
    for (int i = 0; i < maze->rows; i++) {
        free(maze->grid[i]);
    }
    free(maze->grid);
    free(maze->path);
    free(maze);
}
```

**关键点**：

- 分配顺序：结构体 → 行指针 → 每行格子 → path 数组
- 释放顺序恰好反过来：每行 → 行指针 → path → 结构体
- `path` 按 `rows × cols` 分配，最坏情况下迷宫是蛇形通路，走遍每一格

## 二、随机化 DFS 生成 — generate.c

### 链栈操作

栈用哑头节点设计——哑头不存数据，真正的数据从 `top->next` 开始：

```c
StackNode *initStack() {
    StackNode *s = (StackNode *)malloc(sizeof(StackNode));
    s->next = NULL;
    return s;
}

void push(StackNode *top, Position pos) {
    StackNode *p = (StackNode *)malloc(sizeof(StackNode));
    p->pos = pos;
    p->next = top->next;   // 新节点插在哑头后面（头插法）
    top->next = p;
}

void pop(StackNode *top, Position *pos) {
    if (top == NULL || top->next == NULL) return;
    StackNode *del = top->next;
    *pos = del->pos;
    top->next = del->next;
    free(del);
}

Position peek(StackNode *top) {
    return top->next->pos;  // 栈顶 = 哑头的下一个
}

bool isStackEmpty(StackNode *top) {
    return top->next == NULL;
}
```

**为什么用哑头**：push/pop 只改哑头的 next，不需要传二级指针。`top` 永远指向哑头不变。

### DFS 生成主算法

```c
void generateRandomizedMaze(Maze *maze) {
    StackNode *stack = initStack();

    // 起点 (0,0) 入栈并标记已访问
    maze->grid[0][0].visited = true;
    Position start = {0, 0};
    push(stack, start);

    // 偏移数组 — 成对描述四个方向的行列变化
    int dRows[] = {-1, 1, 0, 0};   // 上、下、左、右
    int dCols[] = {0, 0, -1, 1};

    while (!isStackEmpty(stack)) {
        Position curr = peek(stack);   // 看栈顶，不出栈

        // === 第 1 步：收集未访问邻居 ===
        int unvisited[4] = {0, 1, 2, 3};
        int count = 0;
        int nextRow, nextCol;

        for (int i = 0; i < 4; i++) {
            nextRow = curr.row + dRows[i];
            nextCol = curr.col + dCols[i];
            if (nextRow >= 0 && nextRow < maze->rows &&
                nextCol >= 0 && nextCol < maze->cols) {
                if (!maze->grid[nextRow][nextCol].visited) {
                    unvisited[count++] = i;  // 存方向编号
                }
            }
        }

        // === 第 2 步：死胡同则回溯 ===
        if (!count) {
            Position tmp;
            pop(stack, &tmp);
            continue;
        }

        // === 第 3 步：Fisher-Yates 洗牌，随机选邻居 ===
        for (int i = count - 1; i > 0; i--) {
            int j = rand() % (i + 1);
            int tmp = unvisited[j];
            unvisited[j] = unvisited[i];
            unvisited[i] = tmp;
        }
        int chosen = unvisited[0];

        // ★ 重新计算邻居坐标（洗牌后的方向可能与循环末次不同）
        nextRow = curr.row + dRows[chosen];
        nextCol = curr.col + dCols[chosen];

        // === 第 4 步：拆墙（两格各改一次） ===
        switch (chosen) {
            case 0: // 上 — 拆当前格的上墙和邻居的下墙
                maze->grid[curr.row][curr.col].top = false;
                maze->grid[nextRow][nextCol].bottom = false;
                break;
            case 1: // 下
                maze->grid[curr.row][curr.col].bottom = false;
                maze->grid[nextRow][nextCol].top = false;
                break;
            case 2: // 左
                maze->grid[curr.row][curr.col].left = false;
                maze->grid[nextRow][nextCol].right = false;
                break;
            case 3: // 右
                maze->grid[curr.row][curr.col].right = false;
                maze->grid[nextRow][nextCol].left = false;
                break;
        }

        // === 第 5 步：标记邻居已访问并入栈 ===
        maze->grid[nextRow][nextCol].visited = true;
        Position next = {nextRow, nextCol};
        push(stack, next);
    }
}
```

**核心逻辑示意**：

```
(0,0) → 收集未访问邻居 → 随机选一个 → 拆墙 → push 邻居
                                                    ↓
                                           当前格变成邻居，重复
                                                    ↓
                                   四个方向全访问过 → pop 回溯
                                                    ↓
                                           找栈顶其他方向继续挖
                                                    ↓
                                     栈空 → 所有格子都访问 → 完成
```

**容易踩的坑**：

- `nextRow`/`nextCol` 在 for 循环收集邻居时被不断覆盖，循环结束留在方向 3（右）。洗牌选中 `chosen` 后，**必须用 `dRows[chosen]`/`dCols[chosen]` 重新算一遍**，否则拆的是错位的墙
- 拆墙要对陈：当前格的上墙拆了，邻居的下墙也得拆——双向打通
- Fisher-Yates 洗牌范围是 `count`（候选数量），不是 `4`
- `srand(time(NULL))` 放在 `main()` 开头只播一次种，多次播种会导致随机序列相同

## 三、BFS 寻路 — solver.c

### 链队操作

队列入队在尾部（rear）、出队在头部（front），FIFO：

```c
Queue *initQueue() {
    Queue *q = (Queue *)malloc(sizeof(Queue));
    q->front = NULL;
    q->rear = NULL;
    return q;
}

void enqueue(Queue *q, QueueNode node) {
    QueueNode *tmp = (QueueNode *)malloc(sizeof(QueueNode));
    *tmp = node;
    tmp->next = NULL;

    if (q->front == NULL) {     // 空队列：头尾指向同一节点
        q->front = tmp;
        q->rear = tmp;
    } else {                    // 非空：尾插
        q->rear->next = tmp;
        q->rear = tmp;
    }
}

void dequeue(Queue *q, QueueNode *node) {
    if (q->front == NULL) return;

    QueueNode *tmp = q->front;
    *node = *tmp;
    q->front = q->front->next;
    if (q->front == NULL) {     // 删完后队列为空，rear 也要清
        q->rear = NULL;
    }
    free(tmp);
}

bool isQueueEmpty(Queue *q) {
    return q->front == NULL;
}
```

**链栈 vs 链队对比**：

| | 链栈（DFS 生成） | 链队（BFS 寻路） |
|---|---|---|
| 操作 | push 头插 / pop 头删 | enqueue 尾插 / dequeue 头删 |
| 指针 | 一个 top（哑头） | 两个 front + rear |
| 哑头 | 有（避免二级指针） | 无（头尾都在变，哑头没用） |
| 顺序 | LIFO（后入先出） | FIFO（先入先出） |

### BFS 主算法

```c
void mazeSolver(Maze *maze) {
    // === 1. 重置 visited（DFS 生成完 visited 全为 true） ===
    for (int i = 0; i < maze->rows; i++)
        for (int j = 0; j < maze->cols; j++)
            maze->grid[i][j].visited = false;

    // === 2. 创建队列，起点入队 ===
    Queue* queue = initQueue();
    QueueNode node;
    node.row = 0; node.col = 0;
    node.next = NULL;
    node.prevRow = -1; node.prevCol = -1;   // -1 = 起点没有前驱
    enqueue(queue, node);
    maze->grid[0][0].visited = true;

    // === 3. 偏移数组（和生成阶段共用同一个思路） ===
    int dRows[] = {-1, 1, 0, 0};
    int dCols[] = {0, 0, -1, 1};

    // === 4. 分配 prev 回溯数组（存每个格子的"上一步"） ===
    int **prevRow = (int **)malloc(sizeof(int *) * maze->rows);
    int **prevCol = (int **)malloc(sizeof(int *) * maze->rows);
    for (int i = 0; i < maze->rows; i++) {
        prevRow[i] = (int *)malloc(sizeof(int) * maze->cols);
        prevCol[i] = (int *)malloc(sizeof(int) * maze->cols);
    }
    for (int i = 0; i < maze->rows; i++)
        for (int j = 0; j < maze->cols; j++)
            prevRow[i][j] = prevCol[i][j] = -1;  // -1 表示无前驱

    // === 5. BFS 主循环 ===
    while (!isQueueEmpty(queue)) {
        QueueNode curr;
        dequeue(queue, &curr);

        // 如果当前节点是终点，退出循环
        if (curr.row == maze->rows - 1 && curr.col == maze->cols - 1)
            break;

        int nextRow, nextCol;
        for (int i = 0; i < 4; i++) {
            nextRow = curr.row + dRows[i];
            nextCol = curr.col + dCols[i];

            // 检查墙是否打通（isAccessible）
            bool isAccessible = false;
            switch (i) {
                case 0: if (maze->grid[curr.row][curr.col].top == false)
                            isAccessible = true; break;
                case 1: if (maze->grid[curr.row][curr.col].bottom == false)
                            isAccessible = true; break;
                case 2: if (maze->grid[curr.row][curr.col].left == false)
                            isAccessible = true; break;
                case 3: if (maze->grid[curr.row][curr.col].right == false)
                            isAccessible = true; break;
            }

            // 三重条件：未越界 + 墙打通 + 未访问
            if (nextRow >= 0 && nextRow < maze->rows &&
                nextCol >= 0 && nextCol < maze->cols) {
                if (isAccessible && !maze->grid[nextRow][nextCol].visited) {
                    // 记录前驱（回溯时会用到）
                    prevRow[nextRow][nextCol] = curr.row;
                    prevCol[nextRow][nextCol] = curr.col;
                    maze->grid[nextRow][nextCol].visited = true;

                    // 邻居入队
                    QueueNode neighbour;
                    neighbour.row = nextRow;   neighbour.col = nextCol;
                    neighbour.prevRow = curr.row; neighbour.prevCol = curr.col;
                    neighbour.next = NULL;
                    enqueue(queue, neighbour);
                }
            }
        }
    }

    // === 6. 从终点沿 prev 倒退回起点，标记路径 ===
    int row = maze->rows - 1, col = maze->cols - 1;
    while (row != -1 && col != -1) {
        maze->grid[row][col].isPath = true;
        maze->path[maze->pathLen].row = row;
        maze->path[maze->pathLen].col = col;
        maze->pathLen++;

        int prevR = prevRow[row][col];
        int prevC = prevCol[row][col];
        row = prevR;
        col = prevC;
    }

    // === 7. 释放 prev 数组 ===
    for (int i = 0; i < maze->rows; i++) {
        free(prevRow[i]);
        free(prevCol[i]);
    }
    free(prevRow);
    free(prevCol);
}
```

**BFS 和 DFS 生成的区别**：

| | DFS 生成 | BFS 寻路 |
|---|---|---|
| 数据结构 | 栈（链表） | 队列（链表） |
| 判断能否走 | 只看 visited | visited + 墙是否打通 |
| visited 含义 | 已被挖通 | 已探索过 |
| 随机性 | Fisher-Yates 洗牌 | 无，按固定顺序扩展 |
| 路径 | 生成时随机连通 | 保证最短路径 |
| 为什么要 prev | 不需要 | 需要，用于回溯路径 |

**为什么需要 prevRow/prevCol 二维数组**：

QueueNode 里的 prevRow/prevCol 随节点出队后就被 free 了，路径信息丢失。所以用 prevRow/prevCol **二维数组**持久化存储每个格子的前驱，出队不影响回溯。

## 四、raylib GUI 渲染 — gui.c

```c
#include <raylib.h>
#include "maze.h"

void drawMaze(Maze *maze, Font font) {
    // ==================== 1. 画格子底色 + 墙 ====================
    for (int row = 0; row < maze->rows; row++) {
        for (int col = 0; col < maze->cols; col++) {
            // 格子左上角像素坐标
            int x = OFFSET_X + col * CELL_SIZE;
            int y = OFFSET_Y + row * CELL_SIZE;

            Cell cell = maze->grid[row][col];

            // 格子中心坐标（起点/终点画圆用）
            int cx = x + CELL_SIZE / 2;
            int cy = y + CELL_SIZE / 2;
            int radius = CELL_SIZE / 4;

            // 1a. 先画格子底色
            DrawRectangle(x, y, CELL_SIZE, CELL_SIZE, DARKGRAY);

            // 1b. 起点画绿圆，终点画红圆
            if (col == 0 && row == 0) {
                DrawCircle(cx, cy, radius, GREEN);
            } else if (col == maze->cols - 1 && row == maze->rows - 1) {
                DrawCircle(cx, cy, radius, RED);
            }

            // 1c. 四面墙画在底色上面
            if (cell.top)    DrawLine(x, y,
                                      x + CELL_SIZE, y, WALL_COLOR);
            if (cell.bottom) DrawLine(x, y + CELL_SIZE,
                                      x + CELL_SIZE, y + CELL_SIZE, WALL_COLOR);
            if (cell.left)   DrawLine(x, y,
                                      x, y + CELL_SIZE, WALL_COLOR);
            if (cell.right)  DrawLine(x + CELL_SIZE, y,
                                      x + CELL_SIZE, y + CELL_SIZE, WALL_COLOR);
        }
    }

    // ==================== 2. 画 BFS 最短路径（金色连线） ====================
    for (int i = 0; i < maze->pathLen - 1; i++) {
        // 路径上相邻两格的中心点像素坐标
        int x1 = OFFSET_X + maze->path[i].col * CELL_SIZE + CELL_SIZE / 2;
        int y1 = OFFSET_Y + maze->path[i].row * CELL_SIZE + CELL_SIZE / 2;
        int x2 = OFFSET_X + maze->path[i+1].col * CELL_SIZE + CELL_SIZE / 2;
        int y2 = OFFSET_Y + maze->path[i+1].row * CELL_SIZE + CELL_SIZE / 2;
        // 用加粗的金色线连接两格中心
        DrawLineEx((Vector2){x1, y1}, (Vector2){x2, y2}, 3.0f, GOLD);
    }
}
```

**画图顺序（从底到顶）**：

```
ClearBackground(深灰背景)
  → 每格底色 DrawRectangle(DARKGRAY)
  → 起点/终点小圆点 DrawCircle
  → 四面墙 DrawLine
  → BFS 路径金线 DrawLineEx  ← 最上层，不会被墙盖住
```

**为什么用连线而不是填色**：

路径格子填色会覆盖墙线，看不清通道结构。格子中心连线法不盖墙、路径清晰，视觉上像一条金蛇从起点蜿蜒到终点。

## 五、主入口 + 交互 — main.c

```c
#include <stdlib.h>
#include <raylib.h>
#include <time.h>
#include "maze.h"

int main(void) {
    srand(time(NULL));                          // 随机种子，只播一次

    InitWindow(LENGTH, WIDTH, "MAZE EXPLORER");
    SetTargetFPS(60);
    Font font = LoadFontEx(FONT_PATH, 20, NULL, 0);

    Maze *maze = createMaze(ROWS, COLS);        // 创建迷宫
    generateRandomizedMaze(maze);               // DFS 生成
    bool solving = false;                       // 防止空格重复触发

    while (!WindowShouldClose()) {
        BeginDrawing();
        ClearBackground(BG_COLOR);
        drawMaze(maze, font);                   // 画迷宫 + 路径
        DrawTextEx(font, "R:Regenerate | Space:Solve | ESC:Quit",
                   (Vector2){10, WIDTH - 25}, 20, 2, GREEN);
        EndDrawing();

        // R 键：重置并重新生成
        if (IsKeyPressed(KEY_R)) {
            for (int row = 0; row < maze->rows; row++) {
                for (int col = 0; col < maze->cols; col++) {
                    maze->grid[row][col].top = true;
                    // ... 其他四面墙和 visited/isPath 全重置
                }
            }
            maze->pathLen = 0;
            solving = false;
            generateRandomizedMaze(maze);
        }

        // 空格键：BFS 寻路（只执行一次，R 后解锁）
        if (IsKeyPressed(KEY_SPACE) && !solving) {
            solving = true;
            mazeSolver(maze);
        }
    }

    destroyMaze(maze);
    UnloadFont(font);
    CloseWindow();
    return 0;
}
```

**交互逻辑**：

| 按键 | 功能 | 防重复机制 |
|------|------|-----------|
| R | 重置并重新 DFS 生成迷宫 | 无限制 |
| Space | 触发 BFS 寻路，显示最短路径 | `solving` 标志位锁 |
| ESC | 退出 | raylib 内置 |

`solving` 锁的作用：Space 触发一次 BFS 后置 true，再按无效。按 R 后锁释放，可以再次寻路。

---

## 算法核心对比

| | DFS 生成 | BFS 寻路 |
|---|---|---|
| 目标 | 挖通所有格子，形成连通迷宫 | 找到起点到终点的最短路径 |
| 数据结构 | 链栈（LIFO，哑头） | 链队（FIFO，front+rear） |
| 判断条件 | 邻居未 visited | 邻居未 visited **且** 中间墙已打通 |
| 随机性 | Fisher-Yates 洗牌选方向 | 固定顺序 0→1→2→3 |
| 最短路径 | 保证不了（只有一条路） | 保证最短 |
| 动画潜力 | 支持逐帧推进 | 支持逐帧推进 |

## 已踩过的坑 (v1.0)

1. **`nextRow`/`nextCol` 值是循环最后一次的方向 3**：洗牌后必须用 `dRows[chosen]` 重新算
2. **`dequeue` 中 `=` vs `==`**：`if (q->front = NULL)` 是赋值不是比较
3. **free 放 while 循环里**：第一轮就把 prev 数组释放了，下一轮 use-after-free 段错误
4. **`srand` 重复播种**：放 `main()` 开头只播一次
5. **R 键重置漏了 `maze->pathLen = 0`**：导致路径残留在画面上
6. **`curr` 声明为未初始化指针**：应用局部变量，传 `&curr`
7. **case 4 vs case 3**：右方向编号是 3，不是 4
